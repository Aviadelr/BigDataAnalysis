#the python code
import json
import sys
from collections import Counter


filename = sys.argv[1]
json_path = '/sci/courses/2022/52002/aviad.elron/' + filename + '.json'
k = int(sys.argv[2])

stop_words_file = "/sci/courses/2022/52002/orzuk/midterm/stop_words.txt"


with open(stop_words_file, "r") as file:
    stop_words = set()
    content = file.read()
    words = content.replace("{", "").replace("}", "").replace('"', '').split(", ")
    words = [word[1:-1] if word.startswith("'") and word.endswith("'") else word for word in words]
    for word in words:
        stop_words.add(word)

with open(json_path, "r") as file:
    data = [json.loads(line) for line in file]

abstracts = [item.get("abstract", "") for item in data]
abstracts = " ".join(abstracts).replace("\\", "").split()
filtered_abstracts = [x.lower() for x in abstracts if x.lower() not in words]

string_counts = Counter(filtered_abstracts)

most_common_strings = string_counts.most_common(k)

year = filename[6:]
print("Most common words for articles published in ", year)
print(" ")
for string, count in most_common_strings:
    print(f'Word: {string}, number of occurences: {count}')
print(" ")


#set directory
